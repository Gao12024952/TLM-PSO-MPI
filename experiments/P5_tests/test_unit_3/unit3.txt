def worker_loop(local_comm):
    rank = local_comm.Get_rank()
    status = MPI.Status()

    while True:
        # 使用 Probe 先检查消息标签
        local_comm.Probe(source=0, tag=MPI.ANY_TAG, status=status)
        tag = status.Get_tag()

        # 1. 优先处理控制消息
        if tag == 0:
            control_msg = local_comm.recv(source=0, tag=0, status=status)
            if control_msg == "QUIT":
                break
            continue

        # 2. 如果不是控制消息，则按顺序接收数据并直接赋给具体变量
        if rank == 1:
            # 接收: 温度(x), 湿度(y), 气压(z)
            x = local_comm.recv(source=0, tag=1, status=status)
            y = local_comm.recv(source=0, tag=2, status=status)
            z = local_comm.recv(source=0, tag=3, status=status)

            path = []
            if (y > z and (y - z) ** 2 > x * 100) != (y * 6 > z and (y - z) ** 2 > x * 100):
                path.append(1)
            if (y > z and (y - z) ** 2 > x * 100) != (y + 10 > z and (y - z) ** 2 > x * 100):
                path.append(2)
            if (y > z and (y - z) ** 2 > x * 100) != (y > z and (y * 4 - z) ** 2 > x * 100):
                path.append(3)
            if (y > z and (y - z) ** 2 > x * 100) != (y > z and (y - z) * 50 > x * 100):
                path.append(4)
            if (y > z and (y - z) ** 2 > x * 100) != (y > z and (z * 6 - z) ** 2 > x * 100):
                path.append(5)
            if (y > z and (y - z) ** 2 > x * 100) != (y > z and (y - z) ** 2 > x * 6):
                path.append(6)
            if (y > z and (y - z) ** 2 > x * 100) != (y > z and (y - z) ** 2 > (x - 100) * 100):
                path.append(7)
            if (y > z and (y - z) ** 2 > x * 100) != (y > z and (y - z) ** 2 > (y - 100) * 100):
                path.append(8)
            if (y > z and (y - z) ** 2 > x * 100) != (y > z and (y + z) ** 2 > x * 100):
                path.append(9)
            if (y > z and (y - z) ** 2 > x * 100) != (y > z and (y - z) ** 2 > x - 100):
                path.append(10)

            if y > z and (y - z) ** 2 > x * 100:
                type_code = 'B1'

            if (x * (y + z) > (y * z) / 10) != (x * 4 * (y + z) > (y * z) / 10):
                path.append(11)
            if (x * (y + z) > (y * z) / 10) != (x * (y * 3 + z) > (y * z) / 10):
                path.append(12)
            if (x * (y + z) > (y * z) / 10) != (x * (y + z * 3) > (y * z) / 10):
                path.append(13)
            if (x * (y + z) > (y * z) / 10) != (x + (y + z) > (y * z) / 10):
                path.append(14)
            if (x * (y + z) > (y * z) / 10) != (x * (y + x * 1.5) > (y * z) / 10):
                path.append(15)
            if (x * (y + z) > (y * z) / 10) != (x * (y + z) > (y / 5 * z) / 10):
                path.append(16)
            if (x * (y + z) > (y * z) / 10) != (x * (y + z) > (y * z - 30) / 10):
                path.append(17)
            if (x * (y + z) > (y * z) / 10) != (x * (y + z) > (y * z) / 40):
                path.append(18)
            if (x * (y + z) > (y * z) / 10) != (x * (y + z + 10) > (y * z) / 10):
                path.append(19)
            if (x * (y + z) > (y * z) / 10) != ((x + 2) * (y + z) > (y * z) / 10):
                path.append(20)

            if x * (y + z) > (y * z) / 10:
                type_code = 'B2'

            if (z * z - y * y > x * 500 and z > 0) != (z * z * 2 - y * y > x * 500 and z > 0):
                path.append(21)
            if (z * z - y * y > x * 500 and z > 0) != (z * (z + 10) - y * y > x * 500 and z > 0):
                path.append(22)
            if (z * z - y * y > x * 500 and z > 0) != (y * z - y * y > x * 500 and z > 0):
                path.append(23)
            if (z * z - y * y > x * 500 and z > 0) != (z * z + y * y > x * 500 and z > 0):
                path.append(24)
            if (z * z - y * y > x * 500 and z > 0) != (z * z - y * (y - 20) > x * 500 and z > 0):

        elif rank == 2:

            z = local_comm.recv(source=0, tag=2, status=status)

            path = []

            if (w < (z - y) / 5 and z > y) != (w / 1.8 < (z - y) / 5 and z > y):

            if (y ** 3 + z ** 3 < w * 1000) != ((y - 30) ** 2 + z ** 3 < w * 1000):

            if y ** 3 + z ** 3 < w * 1000:
                type_code = 'C2'

            if ((y - w) ** 2 > z + 10) != ((y - w + 1.2) ** 2 > z + 10):
                path.append(21)
            if ((y - w) ** 2 > z + 10) != ((y * 1.8 - w) ** 2 > z + 10):
                path.append(22)
            if ((y - w) ** 2 > z + 10) != ((y - w / 3) ** 2 > z + 10):
                path.append(23)
            if ((y - w) ** 2 > z + 10) != ((y - w) ** 2 > z / 18 + 10):
                path.append(24)
            if ((y - w) ** 2 > z + 10) != ((y - w) ** 2 > y / 50 + 10):
                path.append(25)
            if ((y - w) ** 2 > z + 10) != ((y - w) ** 2 > z + 10 - 1.2 * z):
                path.append(26)
            if ((y - w) ** 2 > z + 10) != ((y - w) ** 2 > -y * 0.1 + 10):
                path.append(27)
            if ((y - w) ** 2 > z + 10) != ((y - z / 6) ** 2 > z + 10):
                path.append(28)
            if ((y - w) ** 2 > z + 10) != ((y - w + 0.6 * y) ** 2 > z + 10):

            if ((y - w) ** 2 > z + 10) != ((y - w) ** 2 > y * 3):
                path.append(30)

            if (y - w) ** 2 > z + 10:
                type_code = 'C3'

            local_comm.send(path, dest=0, tag=21)

        elif rank == 3:

            z = local_comm.recv(source=0, tag=1, status=status)
            w = local_comm.recv(source=0, tag=2, status=status)
            m = local_comm.recv(source=0, tag=3, status=status)

            path = []
            if (m * (z - w) > z * 2) != (m * 2 * (z - w) > z * 2):

            if (m * (z - w) > z * 2) != ((m + 8) * (z - w) > z * 2):

            if z / (m * m) > w * 5:

            if ((m % 2) == 0 and z > w * 10) != ((m % 2) > 0 and z > w * 10):

def generate_mpi_paths(comm, a):
    """主进程与工作进程通信，处理5个变量"""
    status = MPI.Status()
    path = []
    x, y, z, w, m = a[0], a[1], a[2], a[3], a[4]

    comm.send(x, dest=1, tag=1)
    comm.send(y, dest=1, tag=2)
    comm.send(z, dest=1, tag=3)

    comm.send(y, dest=2, tag=1)
    comm.send(z, dest=2, tag=2)
    comm.send(w, dest=2, tag=3)

    comm.send(z, dest=3, tag=1)
    comm.send(w, dest=3, tag=2)
    comm.send(m, dest=3, tag=3)

    # 主进程的路径计算
    if (x ** m > (y + z) / 20) != (x ** (m + 15) > (y + z) / 20):
        path.append(1)
    if (x ** m > (y + z) / 20) != (x ** m > (y + z / 2) / 20):
        path.append(2)
    if (x ** m > (y + z) / 20) != (x ** m > (y / 3 + z) / 20):
        path.append(3)
    if (x ** m > (y + z) / 20) != (x ** m > (y + z) / 50):
        path.append(4)
    if (x ** m > (y + z) / 20) != (x ** m > (-y / 18 + z) / 20):
        path.append(5)
    if (x ** m > (y + z) / 20) != (x ** (5 * x) > (y + z) / 20):
        path.append(6)
    if (x ** m > (y + z) / 20) != (x ** (y + 10) > (y + z) / 20):
        path.append(7)
    if (x ** m > (y + z) / 20) != (x ** m > (y + z) / (20 + 55 * x)):
        path.append(8)
    if (x ** m > (y + z) / 20) != (x ** m > (y + z - 5) / 20):
        path.append(9)
    if (x ** m > (y + z) / 20) != (x ** m > (y + z - y / 1.5) / 20):
        path.append(10)

    if x ** m > (y + z) / 20:
        type_code = 'A1'

    if (y * w * m > z * x * 100) != ((y + 10) * w * m > z * x * 100):
        path.append(11)
    if (y * w * m > z * x * 100) != (y * (w + 10) * m > z * x * 100):
        path.append(12)
    if (y * w * m > z * x * 100) != (y * w * (m + 20) > z * x * 100):
        path.append(13)
    if (y * w * m > z * x * 100) != (y * w * m * 4.8 > z * x * 100):
        path.append(14)
    if (y * w * m > z * x * 100) != (y * w * m > (z - 3) * x * 100):
        path.append(15)
    if (y * w * m > z * x * 100) != (y * w * m > z * (x + 105) * 100):
        path.append(16)
    if (y * w * m > z * x * 100) != (y * w * m > z * x * 25):
        path.append(17)
    if (y * w * m > z * x * 100) != (y * w * m > z * x * (100 - 110 * x)):
        path.append(18)
    if (y * w * m > z * x * 100) != (y * w * m > z * x + 100):
        path.append(19)
    if (y * w * m > z * x * 100) != (y * w * m > z * x * (z + 10)):
        path.append(20)

    if y * w * m > z * x * 100:
        type_code = 'A2'

    if (m > 3 and y + z > w * 20) != (m > 3 and y + z > (w - 1) * 20):
        path.append(21)
    if (m > 3 and y + z > w * 20) != (m > 3 and y + y * 8 > w * 20):
        path.append(22)

        path.append(26)
    if (m > 3 and y + z > w * 20) != (m > 3 and y * z > w * 20):
        path.append(27)
    if (m > 3 and y + z > w * 20) != (m > 3 and y + w * 18 > w * 20):
        path.append(28)
    if (m > 3 and y + z > w * 20) != (m > 3 and 3 * z + z > w * 20):
        path.append(29)
    if (m > 3 and y + z > w * 20) != (m > 3 and y + z > w * w * 2):
        path.append(30)

    if m > 3 and y + z > w * 20:
        type_code = 'A3'

    path1 = comm.recv(source=1, tag=11, status=status)
    path2 = comm.recv(source=2, tag=21, status=status)
    path3 = comm.recv(source=3, tag=31, status=status)

    return [path, path1, path2, path3]
