def worker_loop(local_comm):
    rank = local_comm.Get_rank()
    status = MPI.Status()

    while True:
        # 使用 Probe 先检查消息标签
        local_comm.Probe(source=0, tag=MPI.ANY_TAG, status=status)
        tag = status.Get_tag()

        # 1. 优先处理控制消息
        if tag == 0:
            control_msg = local_comm.recv(source=0, tag=0, status=status)
            if control_msg == "QUIT":
                break
            continue

        # 2. 如果不是控制消息，则按顺序接收数据并直接赋给具体变量
        if rank == 1:

            z = local_comm.recv(source=0, tag=3, status=status)

            path = []
            if (x * z > y + 200) != (x * 1.8 * z > y + 200):
                path.append(1)
            if (x * z > y + 200) != (x * (z + 8) > y + 200):
                path.append(2)
            if (x * z > y + 200) != ((x + 12) * z > y + 200):
                path.append(3)
            if (x * z > y + 200) != (x * z > y / 2 + 200):
                path.append(4)
            if (x * z > y + 200) != (26 * z > y + 200):

            if (x * z > y + 200) != (x * z > y + 75):

            if (x * z > y + 200) != (x * z > x * 5 + 200):
                path.append(7)
            if (x * z > y + 200) != ((z + 10) * z > y + 200):
                path.append(8)
            if (x * z > y + 200) != (x * z > y - x * 5.8 + 200):
                path.append(9)
            if (x * z > y + 200) != (x * z > 100 + 200):
                path.append(10)

            if x * z > y + 200:
                type_code = 'B1'

            if (y / x > z * 2 + 15) != (y * 2 / x > z * 2 + 15):
                path.append(11)
            if (y / x > z * 2 + 15) != ((y + 300) / x > z * 2 + 15):
                path.append(12)
            if (y / x > z * 2 + 15) != (y / x * 2 > z * 2 + 15):
                path.append(13)
            if (y / x > z * 2 + 15) != (y / x > z * 0.2 + 15):
                path.append(14)
            if (y / x > z * 2 + 15) != (y / x > z * 2):
                path.append(15)
            if (y / x > z * 2 + 15) != (x * 30 / x > z * 2 + 15):

                type_code = 'B2'

            if (x + y / 8 < z ** 2) != (x * 5 + y / 8 < z ** 2):

def generate_mpi_paths(comm, a):

    status = MPI.Status()
    path = []
    x, y, z, w, m = a[0], a[1], a[2], a[3], a[4]

    comm.send(x, dest=1, tag=1)
    comm.send(y, dest=1, tag=2)
    comm.send(z, dest=1, tag=3)

    comm.send(y, dest=2, tag=1)
    comm.send(z, dest=2, tag=2)
    comm.send(w, dest=2, tag=3)

    comm.send(z, dest=3, tag=1)
