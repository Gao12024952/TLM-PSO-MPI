def worker_loop(local_comm):
    rank = local_comm.Get_rank()
    status = MPI.Status()

    while True:
        # 使用 Probe 先检查消息标签
        local_comm.Probe(source=0, tag=MPI.ANY_TAG, status=status)
        tag = status.Get_tag()

        # 1. 优先处理控制消息
        if tag == 0:
            control_msg = local_comm.recv(source=0, tag=0, status=status)
            if control_msg == "QUIT":
                break
            continue

        # 2. 如果不是控制消息，则按顺序接收数据并直接赋给具体变量
        if rank == 1:
            # 接收: 温度(x), 湿度(y), 气压(z)
            x = local_comm.recv(source=0, tag=1, status=status)
            y = local_comm.recv(source=0, tag=2, status=status)
            z = local_comm.recv(source=0, tag=3, status=status)

            path = []
            if (z * x > y ** 2 / 3) != (z * x * 2.5 > y ** 2 / 3):
                path.append(1)
            if (z * x > y ** 2 / 3) != (z * (x + 80) > y ** 2 / 3):
                path.append(2)
            if (z * x > y ** 2 / 3) != ((z + 80) * x > y ** 2 / 3):
                path.append(3)
            if (z * x > y ** 2 / 3) != (z * x > y ** 2 / 8):
                path.append(4)
            if (z * x > y ** 2 / 3) != (z * x > (y - 80) ** 2 / 3):
                path.append(5)
            if (z * x > y ** 2 / 3) != (z * 150 > y ** 2 / 3):
                path.append(6)
            if (z * x > y ** 2 / 3) != (180 * x > y ** 2 / 3):
                path.append(7)
            if (z * x > y ** 2 / 3) != (z * x > y * (y - 100) / 3):
                path.append(8)
            if (z * x > y ** 2 / 3) != (z * x > y * 80 / 3):
                path.append(9)
            if (z * x > y ** 2 / 3) != (z * y / 1.2 > y ** 2 / 3):
                path.append(10)

            if z * x > y ** 2 / 3:
                type_code = 'B1'

            if (x / z < y / 20) != (x * 4 / z < y / 20):
                path.append(11)
            if (x / z < y / 20) != ((x + 200) / z < y / 20):
                path.append(12)
            if (x / z < y / 20) != (250 / z < y / 20):
                path.append(13)
            if (x / z < y / 20) != (x / 10 < y / 20):
                path.append(14)
            if (x / z < y / 20) != (x / z < y / 80):
                path.append(15)
            if (x / z < y / 20) != (x / z < (y - 80) / 20):
                path.append(16)
            if (x / z < y / 20) != (x / z < 50 / 20):
                path.append(17)
            if (x / z < y / 20) != ((y + 50) / z < y / 20):
                path.append(18)
            if (x / z < y / 20) != ((z + 180) / z < y / 20):
                path.append(19)
            if (x / z < y / 20) != (x / z + 5 < y / 20):
                path.append(20)

            if x / z < y / 20:
                type_code = 'B2'

            if (x ** 2 + z ** 2 < y * 5) != ((x / 5) ** 2 + z ** 2 < y * 5):
                path.append(21)
            if (x ** 2 + z ** 2 < y * 5) != (15 ** 2 + z ** 2 < y * 5):
                path.append(22)
            if (x ** 2 + z ** 2 < y * 5) != (x ** 1.2 + z ** 2 < y * 5):
                path.append(23)
            if (x ** 2 + z ** 2 < y * 5) != (x ** 2 + (z - 60) ** 2 < y * 5):
                path.append(24)
            if (x ** 2 + z ** 2 < y * 5) != (x ** 2 + (z / 4) ** 2 < y * 5):
                path.append(25)
            if (x ** 2 + z ** 2 < y * 5) != (x ** 2 + z * (z - 60) < y * 5):
                path.append(26)
            if (x ** 2 + z ** 2 < y * 5) != (x * (x - 60) + z ** 2 < y * 5):
                path.append(27)
            if (x ** 2 + z ** 2 < y * 5) != (x * 5 + z ** 2 < y * 5):
                path.append(28)
            if (x ** 2 + z ** 2 < y * 5) != (x * z / 2 + z ** 2 < y * 5):
                path.append(29)
            if (x ** 2 + z ** 2 < y * 5) != (x ** 2 + z ** 2 < y * 5 + 2000):
                path.append(30)

            if x ** 2 + z ** 2 < y * 5:
                type_code = 'B3'

            local_comm.send(path, dest=0, tag=11)

        elif rank == 2:
            # 接收: 湿度(y), 气压(z), 风速(w)
            y = local_comm.recv(source=0, tag=1, status=status)
            z = local_comm.recv(source=0, tag=2, status=status)

            if ((y - z) ** 2 > w * 100) != ((y / 2 - z) ** 2 > w * 100):

            if (y - z) ** 2 > w * 100:
                type_code = 'C3'

        elif rank == 3:

            z = local_comm.recv(source=0, tag=1, status=status)
            w = local_comm.recv(source=0, tag=2, status=status)
            m = local_comm.recv(source=0, tag=3, status=status)

            path = []
            if (m * z / 2 > w ** 2 + 50) != (m * 2.5 * z / 2 > w ** 2 + 50):
                path.append(1)
            if (m * z / 2 > w ** 2 + 50) != (m * (z + 80) / 2 > w ** 2 + 50):
                path.append(2)
            if (m * z / 2 > w ** 2 + 50) != ((m + 10) * z / 2 > w ** 2 + 50):
                path.append(3)
            if (m * z / 2 > w ** 2 + 50) != (m * z * 1.5 > w ** 2 + 50):
                path.append(4)
            if (m * z / 2 > w ** 2 + 50) != (m * z / 2 > w ** 2 + 300):

            if (z ** 2 + w ** 2 > m ** 3 + 500) != ((z + 50) ** 2 + w ** 2 > m ** 3 + 500):

                type_code = 'D3'

            local_comm.send(path, dest=0, tag=31)


def generate_mpi_paths(comm, a):
    """主进程与工作进程通信，处理5个变量"""
    status = MPI.Status()
    path = []
    x, y, z, w, m = a[0], a[1], a[2], a[3], a[4]

    comm.send(x, dest=1, tag=1)
    comm.send(y, dest=1, tag=2)
    comm.send(z, dest=1, tag=3)

    comm.send(y, dest=2, tag=1)
    comm.send(z, dest=2, tag=2)
    comm.send(w, dest=2, tag=3)

    comm.send(z, dest=3, tag=1)
    comm.send(w, dest=3, tag=2)
    comm.send(m, dest=3, tag=3)

    # 主进程的路径计算
