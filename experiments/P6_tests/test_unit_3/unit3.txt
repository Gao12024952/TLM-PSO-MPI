def worker_loop(local_comm):
    rank = local_comm.Get_rank()
    status = MPI.Status()

    while True:
        # 使用 Probe 先检查消息标签
        local_comm.Probe(source=0, tag=MPI.ANY_TAG, status=status)
        tag = status.Get_tag()

        # 1. 优先处理控制消息
        if tag == 0:
            control_msg = local_comm.recv(source=0, tag=0, status=status)
            if control_msg == "QUIT":
                break
            continue

        # 2. 如果不是控制消息，则按顺序接收数据并直接赋给具体变量
        if rank == 1:
            # 接收: 温度(x), 湿度(y), 气压(z)
            x = local_comm.recv(source=0, tag=1, status=status)
            y = local_comm.recv(source=0, tag=2, status=status)
            z = local_comm.recv(source=0, tag=3, status=status)

            path = []
            if (x ** 3 + y > z * x * 15) != ((x * 1.8) ** 3 + y > z * x * 15):
                path.append(1)
            if (x ** 3 + y > z * x * 15) != (x ** 3 + y + 620 > z * x * 15):
                path.append(2)
            if (x ** 3 + y > z * x * 15) != (x ** 3.8 + y > z * x * 15):
                path.append(3)
            if (x ** 3 + y > z * x * 15) != (x ** 3 + y * 15 > z * x * 15):
                path.append(4)
            if (x ** 3 + y > z * x * 15) != (x ** 3 + y > z * x * 5):
                path.append(5)
            if (x ** 3 + y > z * x * 15) != (x ** 3 + y > z * (x - 5) * 15):
                path.append(6)
            if (x ** 3 + y > z * x * 15) != (x ** 3 + y > (z - 10) * x * 15):
                path.append(7)
            if (x ** 3 + y > z * x * 15) != (x ** 3 + y > z * 55 * 15):
                path.append(8)
            if (x ** 3 + y > z * x * 15) != (x ** 3 + y > 20 * x * 15):
                path.append(9)
            if (x ** 3 + y > z * x * 15) != (x ** 3 + 650 > z * x * 15):
                path.append(10)

            if x ** 3 + y > z * x * 15:
                type_code = 'B1'

            if (y % (z + 5) < x / 4) != (y % (z + 5) * 1.8 < x / 4):
                path.append(11)
            if (y % (z + 5) < x / 4) != (y % (z + 5) < (x - 10) / 4):
                path.append(12)
            if (y % (z + 5) < x / 4) != (y % (z + 5) < x / (x / 5 + 1)):
                path.append(13)
            if (y % (z + 5) < x / 4) != (y % (z + 5) < 5.9):
                path.append(14)
            if (y % (z + 5) < x / 4) != (y % (z + 5) < x / 6):
                path.append(15)
            if (y % (z + 5) < x / 4) != (y % (z + 5) < (x - 10) / 4):
                path.append(16)
            if (y % (z + 5) < x / 4) != (y % (z + 5) < x / (z / 6 + 1)):
                path.append(17)
            if (y % (z + 5) < x / 4) != (y % (z + 5) < x / (y / 8 + 1)):
                path.append(18)
            if (y % (z + 5) < x / 4) != (y % (z + 5) < 7.44):
                path.append(19)
            if (y % (z + 5) < x / 4) != ((y / 2) % (z + 5) < x / 4):
                path.append(20)

            if y % (z + 5) < x / 4:
                type_code = 'B2'

            if ((y - x) * z > y ** 2 / 8) != ((y * 1.2 - x) * z > y ** 2 / 8):
                path.append(21)
            if ((y - x) * z > y ** 2 / 8) != ((y - x / 1.8) * z > y ** 2 / 8):
                path.append(22)
            if ((y - x) * z > y ** 2 / 8) != ((y - x) * z * 1.8 > y ** 2 / 8):
                path.append(23)
            if ((y - x) * z > y ** 2 / 8) != ((y - x) * z > y ** 2 / 12):
                path.append(24)
            if ((y - x) * z > y ** 2 / 8) != ((y + 10 - x) * z > y ** 2 / 8):
                path.append(25)
            if ((y - x) * z > y ** 2 / 8) != ((y - x) * (z + 5) > y ** 2 / 8):
                path.append(26)
            if ((y - x) * z > y ** 2 / 8) != ((y - x) * 8 > y ** 2 / 8):
                path.append(27)
            if ((y - x) * z > y ** 2 / 8) != ((y - x) * z > y * (x + 20) / 8):
                path.append(28)
            if ((y - x) * z > y ** 2 / 8) != ((y - x) * z > (y - 10) ** 2 / 8):
                path.append(29)
            if ((y - x) * z > y ** 2 / 8) != ((y - x) * z > y * (y - 20) / 8):
                path.append(30)

            if (y - x) * z > y ** 2 / 8:
                type_code = 'B3'

            local_comm.send(path, dest=0, tag=11)

        elif rank == 2:
            # 接收: 湿度(y), 气压(z), 风速(w)
            y = local_comm.recv(source=0, tag=1, status=status)
            z = local_comm.recv(source=0, tag=2, status=status)

            path = []

            if (w ** 2 / 50 < y * z) != ((w * 1.8) ** 2 / 50 < y * z):

            if (w ** 2 / 50 < y * z) != (w ** 2 / 50 + 80 < y * z):
                path.append(10)

            if w ** 2 / 50 < y * z:
                type_code = 'C1'

            if ((y + z * 6) ** 2 > w * 15) != ((y / 20 + z * 6) ** 2 > w * 15):

            if ((y + z * 6) ** 2 > w * 15) != ((y + z * 6) ** 2 > 350 * 15):
                path.append(20)

            if (y + z * 6) ** 2 > w * 15:
                type_code = 'C2'

            if (z * w < y ** 2 + z * 12) != (z * 2.8 * w < y ** 2 + z * 12):
                path.append(21)
            if (z * w < y ** 2 + z * 12) != (z * (w + 50) < y ** 2 + z * 12):
                path.append(22)
            if (z * w < y ** 2 + z * 12) != ((z + 20) * w < y ** 2 + z * 12):
                path.append(23)
            if (z * w < y ** 2 + z * 12) != (z * w < y ** 1.72 + z * 12):
                path.append(24)
            if (z * w < y ** 2 + z * 12) != (z * w < y * (y - 20) + z * 12):
                path.append(25)
            if (z * w < y ** 2 + z * 12) != (30 * w < y ** 2 + z * 12):
                path.append(26)
            if (z * w < y ** 2 + z * 12) != (z * 15 < y ** 2 + z * 12):
                path.append(27)
            if (z * w < y ** 2 + z * 12) != (z * w < y * 80 + z * 12):
                path.append(28)
            if (z * w < y ** 2 + z * 12) != (z * w < y * (z + 50) + z * 12):
                path.append(29)
            if (z * w < y ** 2 + z * 12) != (z * y < y ** 2 + z * 12):
                path.append(30)

            if z * w < y ** 2 + z * 12:
                type_code = 'C3'

            local_comm.send(path, dest=0, tag=21)

        elif rank == 3:
            # 接收: 气压(z), 风速(w), 日照(m)
            z = local_comm.recv(source=0, tag=1, status=status)
            w = local_comm.recv(source=0, tag=2, status=status)
            m = local_comm.recv(source=0, tag=3, status=status)

            path = []
            if (m ** 2 + w / 5 > z * 25) != ((m * 1.8) ** 2 + w / 5 > z * 25):
                path.append(1)
            if (m ** 2 + w / 5 > z * 25) != ((m + 5) ** 2 + w / 5 > z * 25):

            if ((z * m) % 15 > w / 10) != ((z * m) % 15 > w % 10):

            if (z * m) % 15 > w / 10:

            if (w * m / 3 < z ** 3 - 200) != (w * m / 3 < (z - 4) ** 3 - 200):
                path.append(25)
            if (w * m / 3 < z ** 3 - 200) != (w * m / 3 < z ** 2.5 - 200):

                path.append(30)

            if w * m / 3 < z ** 3 - 200:
                type_code = 'D3'

            local_comm.send(path, dest=0, tag=31)


def generate_mpi_paths(comm, a):
    """主进程与工作进程通信，处理5个变量"""
    status = MPI.Status()
    path = []
    x, y, z, w, m = a[0], a[1], a[2], a[3], a[4]

    comm.send(x, dest=1, tag=1)
    comm.send(y, dest=1, tag=2)
    comm.send(z, dest=1, tag=3)

    comm.send(y, dest=2, tag=1)
    comm.send(z, dest=2, tag=2)
    comm.send(w, dest=2, tag=3)

    comm.send(z, dest=3, tag=1)
    comm.send(w, dest=3, tag=2)
    comm.send(m, dest=3, tag=3)

    # 主进程的路径计算
    if (x * y * z > w * m + 1500) != (x * y * z * 3 > w * m + 1500):
        path.append(1)
