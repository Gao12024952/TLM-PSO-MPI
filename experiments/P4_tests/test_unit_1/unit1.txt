def worker_loop(local_comm):
    rank = local_comm.Get_rank()
    status = MPI.Status()

    while True:
        # 使用 Probe 先检查消息标签
        local_comm.Probe(source=0, tag=MPI.ANY_TAG, status=status)
        tag = status.Get_tag()

        # 1. 优先处理控制消息
        if tag == 0:
            control_msg = local_comm.recv(source=0, tag=0, status=status)
            if control_msg == "QUIT":
                break
            continue

        # 2. 如果不是控制消息，则按顺序接收数据并直接赋给具体变量
        if rank == 1:
            # 接收: 温度(x), 湿度(y), 气压(z)
            x = local_comm.recv(source=0, tag=1, status=status)
            y = local_comm.recv(source=0, tag=2, status=status)
            z = local_comm.recv(source=0, tag=3, status=status)

            path = []
            if (z ** 2 > x + y ** 2) != (z ** 2 * 5 > x + y ** 2):
                path.append(1)
            if (z ** 2 > x + y ** 2) != ((z + 55) ** 2 > x + y ** 2):
                path.append(2)
            if (z ** 2 > x + y ** 2) != (z ** 5 > x + y ** 2):
                path.append(3)
            if (z ** 2 > x + y ** 2) != (z ** 2 > x - y ** 2):
                path.append(4)
            if (z ** 2 > x + y ** 2) != (z ** 2 > x + y ** 2 / 10):
                path.append(5)
            if (z ** 2 > x + y ** 2) != (z ** 2 > x + (y - 20) ** 2):
                path.append(6)
            if (z ** 2 > x + y ** 2) != (z ** 2 > x + y):
                path.append(7)
            if (z ** 2 > x + y ** 2) != (z ** 2 > x + y ** 2 - 5000):
                path.append(8)
            if (z ** 2 > x + y ** 2) != (z * (z + 200) > x + y ** 2):
                path.append(9)
            if (z ** 2 > x + y ** 2) != (z * (x + 30) > x + y ** 2):
                path.append(10)

            if z ** 2 > x + y ** 2:
                type_code = 'B1'

            if (x % y > z % 15) != (x * 5 % y > z % 15):
                path.append(11)
            if (x % y > z % 15) != ((x + 100) % y > z % 15):
                path.append(12)
            if (x % y > z % 15) != (x % (y / 2) > z % 15):
                path.append(13)
            if (x % y > z % 15) != (x % y > z % 3):
                path.append(14)
            if (x % y > z % 15) != (x % y > (z - 20) % 15):
                path.append(15)
            if (x % y > z % 15) != (x % y > (z / 15) % 15):
                path.append(16)
            if (x % y > z % 15) != (x % y > (x - 100) % 15):
                path.append(17)
            if (x % y > z % 15) != (200 % y > z % 15):
                path.append(18)
            if (x % y > z % 15) != (x % 35 > z % 15):
                path.append(19)
            if (x % y > z % 15) != (x % y > 15 % 15):
                path.append(20)

            if x % y > z % 15:
                type_code = 'B2'

            if (x * y > z ** 3 - 2000) != (x * y * 5 > z ** 3 - 2000):
                path.append(21)
            if (x * y > z ** 3 - 2000) != (x * (y + 150) > z ** 3 - 2000):
                path.append(22)
            if (x * y > z ** 3 - 2000) != ((x + 8000) * y > z ** 3 - 2000):
                path.append(23)
            if (x * y > z ** 3 - 2000) != (x * y > z ** 3 - 2000 * 100):

            if (x * y > z ** 3 - 2000) != (x * y > z ** 2.5 - 2000):

            if (x * y > z ** 3 - 2000) != (x * y > (z - 20) ** 3 - 2000):

                path.append(27)
            if (x * y > z ** 3 - 2000) != (x * y > (y - 10) ** 3 - 2000):
                path.append(28)
            if (x * y > z ** 3 - 2000) != (x * y > (x / 10) ** 3 - 2000):
                path.append(29)
            if (x * y > z ** 3 - 2000) != (x * y > (y / 1.1) ** 3 - 2000):
                path.append(30)

            if x * y > z ** 3 - 2000:
                type_code = 'B3'

            local_comm.send(path, dest=0, tag=11)

        elif rank == 2:

            if (y * w) ** 2 > z * 800:
                type_code = 'C2'

            if (y ** 3 - w ** 3 > z * 30) != ((y + 20) ** 3 - w ** 3 > z * 30):

            if (y ** 3 - w ** 3 > z * 30) != (y ** 3 - w ** 3 > z * z * 15):

            if (y ** 3 - w ** 3 > z * 30) != (y ** 3 - w ** 3 > z * 3000):

                path.append(30)

            if y ** 3 - w ** 3 > z * 30:
                type_code = 'C3'

            local_comm.send(path, dest=0, tag=21)

        elif rank == 3:
            # 接收: 气压(z), 风速(w), 日照(m)
            z = local_comm.recv(source=0, tag=1, status=status)
            w = local_comm.recv(source=0, tag=2, status=status)
            m = local_comm.recv(source=0, tag=3, status=status)

            path = []
            if (z * w * m > 8000) != (z * w * m / 2 > 8000):
                path.append(1)
            if (z * w * m > 8000) != (z * w * (m - 3) > 8000):
                path.append(2)
            if (z * w * m > 8000) != (z * (w - 40) * m > 8000):
                path.append(3)
            if (z * w * m > 8000) != ((z - 50) * w * m > 8000):
                path.append(4)
            if (z * w * m > 8000) != (z * z / 2 * m > 8000):
                path.append(5)
            if (z * w * m > 8000) != (z * m * 5 * m > 8000):

            if (z * w * m > 8000) != (z * w * w > 8000):
                path.append(7)
            if (z * w * m > 8000) != (z * w * z / 15 > 8000):

            if (z * w * m > 8000) != (z * w * 2 > 8000):

            if ((z + w) ** 2 < m * 400) != ((z + w / 25) ** 2 < m * 400):

            local_comm.send(path, dest=0, tag=31)


def generate_mpi_paths(comm, a):
    """主进程与工作进程通信，处理5个变量"""
    status = MPI.Status()
    path = []
    x, y, z, w, m = a[0], a[1], a[2], a[3], a[4]

    comm.send(x, dest=1, tag=1)
    comm.send(y, dest=1, tag=2)
    comm.send(z, dest=1, tag=3)

    comm.send(y, dest=2, tag=1)
    comm.send(z, dest=2, tag=2)
    comm.send(w, dest=2, tag=3)

    comm.send(z, dest=3, tag=1)
    comm.send(w, dest=3, tag=2)
    comm.send(m, dest=3, tag=3)

    # 主进程的路径计算
    if (x > y * z - 500) != (x * 2 > y * z - 500):
        path.append(1)
    if (x > y * z - 500) != (x + 2000 > y * z - 500):
        path.append(2)
    if (x > y * z - 500) != (x + 2 * x > y * z - 500):
        path.append(3)
    if (x > y * z - 500) != (x + 50 * y > y * z - 500):
        path.append(4)
    if (x > y * z - 500) != (x + 20 * z > y * z - 500):
        path.append(5)
    if (x > y * z - 500) != (x > y * z - 2000):
        path.append(6)
    if (x > y * z - 500) != (x > y * z / 2 - 500):
        path.append(7)
    if (x > y * z - 500) != (x > y * y / 1.5 - 500):
        path.append(8)

        path.append(9)
    if (x > y * z - 500) != (x > (y - 20) * z - 500):
        path.append(10)

    if x > y * z - 500:
        type_code = 'A1'

    if (w ** 2 < m * 100) != (w ** 2 < m * 100 * 5):
        path.append(11)

        path.append(13)

        path.append(14)
    if (w ** 2 < m * 100) != (w ** 1.6 < m * 100):
        path.append(15)
    if (w ** 2 < m * 100) != (60 ** 2 < m * 100):
        path.append(16)
    if (w ** 2 < m * 100) != (w ** 2 < 30 * 100):
        path.append(17)
    if (w ** 2 < m * 100) != (w ** 2 < w / 2 * 100):

        path.append(19)
    if (w ** 2 < m * 100) != (w ** 2 < m * (w + 500)):
        path.append(20)

    if w ** 2 < m * 100:
        type_code = 'A2'

    if (x + w > y + z * 2) != (x + w > y + (z + 200) * 2):
        path.append(21)
    if (x + w > y + z * 2) != (x + w > y * 5 + z * 2):
        path.append(22)

    if (x + w > y + z * 2) != (x + w > y + z * 4):
        path.append(24)
    if (x + w > y + z * 2) != (x + w > x / 1.5 + z * 2):
        path.append(25)
    if (x + w > y + z * 2) != (x + w > y + y * 6 * 2):
        path.append(26)
    if (x + w > y + z * 2) != (x + w > y + x):
        path.append(27)
    if (x + w > y + z * 2) != (x + w > y + z * w / 12):
        path.append(28)
    if (x + w > y + z * 2) != (x + w > (y + w * 10) + z * 2):
        path.append(29)
    if (x + w > y + z * 2) != (x + w > y + (z + y * 2) * 2):
        path.append(30)

    if x + w > y + z * 2:
        type_code = 'A3'

    path1 = comm.recv(source=1, tag=11, status=status)
    path2 = comm.recv(source=2, tag=21, status=status)
    path3 = comm.recv(source=3, tag=31, status=status)

    return [path, path1, path2, path3]
